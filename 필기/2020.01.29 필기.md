```python
# f-string float

number = 11.33

print(f'{number}')
print(f'{number:.1f}')
print(f'{number:.5f}')
```

```python
# f-string width 

for i in range(1, 11):
    print(f'{i:02}')
```

```python
# f-string align

new_string = 'abc'

print(f'{new_string:>10}')
print(f'{new_string:*>10}')
```
